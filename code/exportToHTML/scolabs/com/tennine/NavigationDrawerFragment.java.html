148      /** 
149       * Users of this fragment must call this method to set up the navigation drawer interactions. 
150       * 
151       * @param fragmentId   The android:id of this fragment in its activity's layout. 
152       * @param drawerLayout The DrawerLayout containing this fragment's UI. 
153       */ 
154      public void setUp(int fragmentId, DrawerLayout drawerLayout) { 
155          mFragmentContainerView = getActivity().findViewById(fragmentId); 
156          mDrawerLayout = drawerLayout; 
157   
158          // set a custom shadow that overlays the main content when the drawer opens 
159          mDrawerLayout.setDrawerShadow(R.drawable.drawer_shadow, GravityCompat.START); 
160          // set up the drawer's list view with items and click listener 
161   
162          ActionBar actionBar = getActionBar(); 
163          actionBar.setDisplayHomeAsUpEnabled(true); 
164          actionBar.setHomeButtonEnabled(true); 
165   
166          // ActionBarDrawerToggle ties together the the proper interactions 
167          // between the navigation drawer and the action bar app icon. 
168          mDrawerToggle = new ActionBarDrawerToggle( 
169                  getActivity(),                    /* host Activity */ 
170                  mDrawerLayout,                    /* DrawerLayout object */ 
171                  R.drawable.ic_drawer,             /* nav drawer image to replace 'Up' caret */ 
172                  R.string.navigation_drawer_open,  /* "open drawer" description for accessibility */ 
173                  R.string.navigation_drawer_close  /* "close drawer" description for accessibility */ 
174          ) { 
175              @Override 
176              public void onDrawerClosed(View drawerView) { 
177                  super.onDrawerClosed(drawerView); 
178                  if (!isAdded()) { 
179                      return; 
180                  } 
181   
182                  getActivity().supportInvalidateOptionsMenu(); // calls onPrepareOptionsMenu() 
183              } 
184   
185              @Override 
186              public void onDrawerOpened(View drawerView) { 
187                  super.onDrawerOpened(drawerView); 
188                  if (!isAdded()) { 
189                      return; 
190                  } 
191   
192                  if (!mUserLearnedDrawer) { 
193                      // The user manually opened the drawer; store this flag to prevent auto-showing 
194                      // the navigation drawer automatically in the future. 
195                      mUserLearnedDrawer = true; 
196                      SharedPreferences sp = PreferenceManager 
197                              .getDefaultSharedPreferences(getActivity()); 
198                      sp.edit().putBoolean(PREF_USER_LEARNED_DRAWER, true).apply(); 
199                  } 
200   
201                  getActivity().supportInvalidateOptionsMenu(); // calls onPrepareOptionsMenu() 
202              } 
203          }; 
204   
205          // If the user hasn't 'learned' about the drawer, open it to introduce them to the drawer, 
206          // per the navigation drawer design guidelines. 
207          if (!mUserLearnedDrawer && !mFromSavedInstanceState) { 
208              mDrawerLayout.openDrawer(mFragmentContainerView); 
209          } 
210   
211          // Defer code dependent on restoration of previous instance state. 
212          mDrawerLayout.post(new Runnable() { 
213              @Override 
214              public void run() { 
215                  mDrawerToggle.syncState(); 
216              } 
217          }); 
218   
219          mDrawerLayout.setDrawerListener(mDrawerToggle); 
220      }

222      private void selectItem(int position) { 
223          mCurrentSelectedPosition = position; 
224          if (mDrawerListView != null) { 
225              mDrawerListView.setItemChecked(position, true); 
226          } 
227          if (mDrawerLayout != null) { 
228              mDrawerLayout.closeDrawer(mFragmentContainerView); 
229          } 
230          if (mCallbacks != null) { 
231              mCallbacks.onNavigationDrawerItemSelected(position); 
232          } 
233      }

264      @Override 
265      public void onCreateOptionsMenu(Menu menu, MenuInflater inflater) { 
266          // If the drawer is open, show the global app actions in the action bar. See also 
267          // showGlobalContextActionBar, which controls the top-left area of the action bar. 
268          if (mDrawerLayout != null && isDrawerOpen()) { 
269              inflater.inflate(R.menu.global, menu); 
270              showGlobalContextActionBar(); 
271          } 
272          super.onCreateOptionsMenu(menu, inflater); 
273      } 
274   
275      @Override 
276      public boolean onOptionsItemSelected(MenuItem item) { 
277          if (mDrawerToggle.onOptionsItemSelected(item)) { 
278              return true; 
279          } 
280   
281          if (item.getItemId() == R.id.action_example) { 
282              Toast.makeText(getActivity(), "Example action.", Toast.LENGTH_SHORT).show(); 
283              return true; 
284          } 
285   
286          return super.onOptionsItemSelected(item); 
287      }

289      /** 
290       * Per the navigation drawer design guidelines, updates the action bar to show the global app 
291       * 'context', rather than just what's in the current screen. 
292       */ 
293      private void showGlobalContextActionBar() { 
294          ActionBar actionBar = getActionBar(); 
295          actionBar.setDisplayShowTitleEnabled(true); 
296          actionBar.setNavigationMode(ActionBar.NAVIGATION_MODE_STANDARD); 
297          actionBar.setTitle(R.string.app_name); 
298      } 
299   
300      private ActionBar getActionBar() { 
301          return ((ActionBarActivity) getActivity()).getSupportActionBar(); 
302      }
